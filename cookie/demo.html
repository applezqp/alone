Cookie

usage

  • Alone.Cookie.set({name:value[,expires:GMTString,path:'',domain:'',secure:'']}[,main])
  • Alone.Cookie.set({name1:value1,name2:value2...[,expires:GMTString,path:'',domain:'',secure:'']}[,main])
  • Alone.Cookie.setSub({name:value[,expires:GMTString,path:'',domain:'',secure:'']},main)
  • Alone.Cookie.setSub({name1:value1,name2:value2...[,expires:GMTString,path:'',domain:'',secure:'']},main)
  • Alone.Cookie.get(name[,main])
  • Alone.Cookie.getSub(name,main)
  • Alone.Cookie.attr({[expires:'',path:''...]},name)
  • Alone.Cookie.del(name[,main])
  • Alone.Cookie.del([name1,name2,name3...][,main])
  • Alone.Cookie.delSub(name,main)
  • Alone.Cookie.delSub([name1,name2,name3...],main)
